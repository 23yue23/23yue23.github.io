apiVersion: extensions/v1beta1
kind: Ingress

name: wordpress-ingress
namespace: default
annotations:
kubernetes.io/ingress.class: “nginx”
spec:
rules:

  • host: wordpress.boshao.wang
    http:
    paths:
  • backend:
    serviceName: wordpress
    servicePort: 80
